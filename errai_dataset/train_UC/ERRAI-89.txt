WSGrid - Focus range becomes corrupt with U-L-D-R, or U-R-D-L
WSGrid. Left-click a cell then use shift and cursor keys to extend selection area. Combination Up, Left, Down, Right or Up, Right, Down, Left lead to highlighted selection area becoming corrupt. The same is not experienced if the combination is Down first then Left, Up, Right or Down, Right, Up and Left. It does not matter if selection causes grid to scroll.

public void onModuleLoad() {

		VerticalPanel vp = new VerticalPanel();

		WSGrid grid = new WSGrid(true, true);
		grid.setTitle("A table");

		for (int x=0;x<COLS; x++) {
			grid.setColumnHeader(0, x, "<b>h"+x+"</b>");
		}
		
		for (int y = 0; y < ROWS; y++) {
			for (int x = 0; x < COLS; x++) {
				grid.setCell(y, x, "(" + x + ", " + y + ")");
			}
		}
		grid.setPreciseHeight(500);
		grid.setPreciseWidth(500);
		vp.add(grid);

		RootPanel.get().add(vp);
		
	}
}
